```typescript
// Phase 1 / Plan 01-01 — Subdomain routing middleware.
//
// D-01 precedence ladder: Host > cookie (x-market) > query (?__market=) > default root.
// D-02: known Host values (hk.*, sg.*) are AUTHORITATIVE and cannot be overridden by cookie/query.
//       This is the security posture encoded by the Plan 01-04 hostile-request Vitest test.
// D-03: unknown hosts (plain *.vercel.app, typos, www.*, bot-set headers, plain localhost:3000) route to (root).
// D-04: NextResponse.rewrite() maps incoming paths to internal /<market>/<path> under App Router route groups
//       — the group names (root)/(hk)/(sg) stay invisible in the URL bar (external URL unchanged).
// D-05: plain localhost:3000 routes to (root) via unknown-host fallthrough (no special-case code path).
//
// Intentional-obsolescence note (CONTEXT.md specifics §1, §2):
// The x-market cookie + ?__market= query bridge is a Phase 1–9 preview-window mechanism because
// Vercel Hobby does not give us wildcard subdomains on *.vercel.app. Once Phase 10 attaches
// proactivsports.com via Cloudflare + binds hk.*/sg.* to Vercel, the bridge becomes unused
// fallback code — but STAYS in middleware as a defensive default for unknown hosts. Do NOT remove
// during the Phase 10 cutover.
//
// Sentry: middleware errors surface automatically via sentry.edge.config.ts (loaded by
// instrumentation.ts register() when NEXT_RUNTIME === "edge"). Do NOT import @sentry/nextjs here.

import { NextResponse, type NextRequest } from "next/server";

type Market = "root" | "hk" | "sg";

const MARKET_COOKIE = "x-market";
const MARKET_QUERY = "__market";
const KNOWN_MARKETS: readonly Market[] = ["root", "hk", "sg"] as const;

/**
 * Resolve the market from the Host header.
 * Returns "hk" for host starting with "hk." (e.g., hk.proactivsports.com, hk.localhost:3000, hk-<sha>.vercel.app).
 * Returns "sg" likewise.
 * Returns "root" for host starting with "root." (dev URL root.localhost:3000).
 * Returns null for unrecognised hosts — caller falls through the precedence ladder.
 *
 * D-02: the prefix match is intentionally exact ("hk." not just "hk") so "hkstudio.proactivsports.com"
 * would NOT match and would fall through to root. Defensive.
 */
function resolveHostMarket(host: string | null): Market | null {
  if (!host) return null;
  // Strip port for localhost dev URLs (hk.localhost:3000 -> hk.localhost).
  const bare = host.split(":")[0].toLowerCase();
  if (bare.startsWith("hk.")) return "hk";
  if (bare.startsWith("sg.")) return "sg";
  if (bare.startsWith("root.")) return "root";
  return null;
}

function isKnownMarket(value: string | undefined): value is Market {
  return value !== undefined && (KNOWN_MARKETS as readonly string[]).includes(value);
}

export function middleware(req: NextRequest): NextResponse {
  const { pathname, search } = req.nextUrl;

  // --- D-01 Step 1: Host is authoritative ---
  const hostMarket = resolveHostMarket(req.headers.get("host"));
  if (hostMarket !== null) {
    // Hostile cookie/query are ignored per D-02.
    return rewriteToMarket(req, hostMarket);
  }

  // --- D-01 Step 2: cookie fallback (preview bridge) ---
  const cookieValue = req.cookies.get(MARKET_COOKIE)?.value;
  if (isKnownMarket(cookieValue)) {
    return rewriteToMarket(req, cookieValue);
  }

  // --- D-01 Step 3: query fallback (preview bridge; sets cookie for subsequent requests) ---
  const queryValue = req.nextUrl.searchParams.get(MARKET_QUERY) ?? undefined;
  if (isKnownMarket(queryValue)) {
    const res = rewriteToMarket(req, queryValue);
    // Claude's Discretion (CONTEXT.md): name=x-market, path=/, SameSite=Lax, Secure only in production.
    // Session cookie (no maxAge) so preview hopping between markets doesn't linger beyond the tab.
    res.cookies.set({
      name: MARKET_COOKIE,
      value: queryValue,
      path: "/",
      sameSite: "lax",
      secure: process.env.VERCEL_ENV === "production",
      httpOnly: false, // client-readable is fine; this is a routing hint, not a secret
    });
    return res;
  }

  // --- D-01 Step 4: default root (D-03 unknown-host fallthrough, D-05 plain localhost) ---
  return rewriteToMarket(req, "root");
}

function rewriteToMarket(req: NextRequest, market: Market): NextResponse {
  // D-04: route groups are invisible in external URLs. We rewrite the internal path only.
  // Incoming pathname "/programmes" with market "hk" -> internal "/hk/programmes".
  // For market "root", rewrite prefix is "/root" — this requires app/(root)/ to be mounted at "/root" via
  // the App Router's route-group mechanism (group names in parens are stripped, so (root) resolves to /).
  // However, rewriting to "/" would cause an infinite rewrite loop when root is the default. So we use
  // "/root" as the internal target and rely on the route-group naming.
  //
  // NOTE: Next.js 15 route groups CANNOT be the rewrite target directly — the rewrite must go to a
  // concrete path. We therefore use the convention where each group has a stub page at the root of its
  // group. When market === "root" and pathname is already "/", we rewrite to "/" (no-op internal path
  // that App Router resolves to app/(root)/page.tsx via route-group matching). For non-root markets we
  // rewrite to "/hk<pathname>" or "/sg<pathname>".
  //
  // IMPLEMENTATION: we prefix the pathname with /{market} for hk/sg, and leave "/" alone for root.
  // The App Router will resolve /hk/programmes to app/(hk)/programmes/page.tsx when such a path exists
  // — for Phase 1 the only page is /, so /hk/ resolves to app/(hk)/page.tsx via the group.
  //
  // To make this work with the flat app/(hk)/page.tsx (no nested /hk directory), we use
  // NextResponse.rewrite with the URL's pathname unchanged for root, and with a /{market} prefix for
  // hk/sg. Next.js 15 resolves /hk to app/(hk)/page.tsx when the group is named (hk).
  //
  // VERIFICATION: see middleware.test.ts (Plan 01-04) — tests assert the x-middleware-rewrite header
  // ends with the correct market prefix.

  const url = req.nextUrl.clone();
  if (market === "root") {
    // Root is the bare path — no prefix. App Router resolves "/" to app/(root)/page.tsx via the group.
    // No rewrite needed for root if the URL is already on a path that (root) handles. But we always
    // call rewrite() so the middleware emits a consistent x-middleware-rewrite header that the test
    // can assert on.
    url.pathname = `/root${url.pathname === "/" ? "" : url.pathname}`;
  } else {
    url.pathname = `/${market}${url.pathname === "/" ? "" : url.pathname}`;
  }
  return NextResponse.rewrite(url);
}

// --- Matcher (Claude's Discretion — exclude non-routable paths) ---
// Excludes: _next/* (Next internals), api/health (future health-check route — not created yet, but reserved),
//           favicon.ico, monitoring (Sentry tunnel per next.config.ts line 45), static assets by extension.
// Rationale: these paths must NEVER route into a market-specific tree even if the Host matches — they are
// either Next internals, Sentry's tunnel (T-MIDDLEWARE-BYPASS mitigation), or static assets.
export const config = {
  matcher: [
    "/((?!_next/|api/health|favicon\\.ico|monitoring|.*\\.(?:svg|png|jpg|jpeg|webp|avif|ico|gif|woff2?)).*)",
  ],
};
```

IMPORTANT — route-group rewrite semantics:

Next.js 15's App Router route groups like `(hk)` are URL-invisible — an internal path of `/hk/anything` does NOT resolve to `app/(hk)/anything/page.tsx`. Route groups group files without adding URL segments. So a rewrite target of `/hk/programmes` will NOT find `app/(hk)/programmes/page.tsx` automatically.

To make this work, create THREE THIN PROXY DIRECTORIES that sit outside the route groups and re-export the group pages:

NO — scratch that. The idiomatic Next.js 15 pattern for per-host routing with route groups is:
1. Each route group's `page.tsx` is mounted at `/` (because group names are invisible).
2. All three groups compete to serve `/` — which is normally a conflict, but with middleware you can pick which one resolves.
3. The trick: use `NextResponse.rewrite()` to set a query param or header that a shared `page.tsx` dispatches on — OR — use explicit folder names in the URL (e.g., `/hk/`) that map to `app/hk/page.tsx` (no parens, no groups).

**DECISION for Phase 1** — use explicit folder paths, not route groups, for middleware rewrites:
- Create `app/(root)/`, `app/(hk)/`, `app/(sg)/` as CONTEXT.md D-04 specifies — they're the source of truth.
- But the middleware rewrites to `/root`, `/hk`, `/sg` (no parens in the URL).
- This requires mapping the URLs `/hk/*` → `app/(hk)/*` somehow.

There is NO clean Next.js 15 primitive for "rewrite /hk to the (hk) route group". The community pattern is: place concrete-path pages (not grouped) and rewrite to those. But CONTEXT.md D-04 explicitly chose route groups.

RESOLUTION — re-read the Next.js 15 middleware + route-group behavior during execution. The working pattern is:

1. Put the three placeholder pages INSIDE route groups: `app/(root)/page.tsx`, `app/(hk)/page.tsx`, `app/(sg)/page.tsx`. All three would naturally resolve to `/` — but Next.js will error on three `/` pages.
2. So instead: put them in explicit folders `app/root/page.tsx`, `app/hk/page.tsx`, `app/sg/page.tsx` (no parens). Middleware rewrites `/` → `/root` (or `/hk` or `/sg`) based on market.
3. The "route groups" terminology in D-04 is semantic shorthand for "three separate placeholder trees." <threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Untrusted HTTP request → middleware.ts | Every incoming request carries a Host header, cookies, and query params that a hostile client can set freely. Middleware is the first hop that interprets these. |
| Middleware → App Router (internal rewrite) | Once middleware rewrites to `/<market>/...`, the App Router trusts the path. Market selection MUST be finalised in middleware before this boundary. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 (T-HOST) | Spoofing | middleware.ts `resolveHostMarket()` | mitigate | D-02 invariant: a known Host value (hk./sg./root.) is the sole routing authority; cookie and query are ignored when Host matches. Enforced by the precedence order in `middleware()` (Host check returns before cookie/query are read). Regression-gated by Plan 01-04 Vitest test (D-16 hostile-request case). |
| T-01-02 (T-COOKIE-FORGE) | Spoofing / Tampering | x-market cookie on preview URLs | accept (ASVS L1 low severity) | On Vercel preview URLs (*.vercel.app, no wildcard subdomains on Hobby), the Host always falls through to "unknown → root", so cookie/query is the fallback routing hint. A forged cookie can only route the attacker's own session to a different market group — and Vercel Deployment Protection (Phase 0 D-14) is the outer access gate. No cross-user impact. Documented as preview-window mechanism; becomes unreachable post-Phase 10 when real subdomains bind. |
| T-01-03 (T-MIDDLEWARE-BYPASS) | Elevation / Information disclosure | middleware.ts `config.matcher` | mitigate | Matcher excludes `_next/`, `api/health`, `favicon.ico`, `monitoring`, and common static asset extensions. The `monitoring` exclusion is critical — Sentry's tunnel route (next.config.ts line 45 `tunnelRoute: "/monitoring"`) must pass through unrewritten. Verified at acceptance: `grep "monitoring" middleware.ts` returns the matcher line. The exclude list does NOT accidentally expose a market route because `/hk/*`, `/sg/*`, `/root/*` are not in the exclude regex. |
| T-01-04 (T-HOST, defensive) | Spoofing (bypass via unknown host variants) | `resolveHostMarket` prefix match | mitigate | Prefix match is EXACT (`bare.startsWith("hk.")`, not `startsWith("hk")`) — so `hkstudio.proactivsports.com`, `hk-foo.vercel.app` without the dot, `hkmalicious.com`, etc., fall through to "unknown → root" rather than matching HK. Host is lowercased + port-stripped before comparison to neutralise case + port manipulation. |
| T-01-05 (T-STUDIO-UNAUTH, Phase 6/10 handoff) | Elevation | app/studio — future phases | transfer / accept for Phase 1 | `/studio` is NOT in Phase 1 scope for this plan (Plan 01-03 mounts Studio).
