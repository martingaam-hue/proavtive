```ts
Sentry.init({
  dsn: process.env.NEXT_PUBLIC_SENTRY_DSN,

  // D-18: tag every event with the Vercel runtime env (production | preview | development)
  environment: process.env.NEXT_PUBLIC_VERCEL_ENV ?? process.env.VERCEL_ENV ?? 'local',

  // D-17: release tagging ties stack-trace source-maps to the exact commit SHA

  // Error-rate sampling (adjust later; 100% fine for Phase 0 low-traffic)
  tracesSampleRate: 1.0,

  // D-20: conservative PII posture
  sendDefaultPii: false,

  // D-20: scrub IP addresses from every event (Sentry defaults to capturing client IP)
  // Setting user.ip_address = '{{auto}}' would capture; omitting it + sendDefaultPii:false is enough.
  // beforeSend scrubs any lingering PII-shaped fields as belt-and-braces.
  beforeSend(event) {
    return scrubPii(event);
  },
});

// PII scrubber — redacts shared field names across request, user, extra, contexts.
// Keep this inline (client-side) so the bundle doesn't pull a helper module.
function scrubPii<T extends { request?: any; user?: any; extra?: any; contexts?: any }>(event: T): T {
  const PII_FIELDS = ['email', 'phone', 'password', 'token', 'authorization', 'cookie', 'childName', 'parentName', 'address'];
  const redact = (obj: unknown): unknown => {
    if (!obj || typeof obj !== 'object') return obj;
    const out: Record<string, unknown> = Array.isArray(obj) ? ([...obj] as unknown as Record<string, unknown>) : { ...(obj as Record<string, unknown>) };
    for (const key of Object.keys(out)) {
      if (PII_FIELDS.some((p) => key.toLowerCase().includes(p))) {
        out[key] = '[redacted]';
      } else if (typeof out[key] === 'object') {
        out[key] = redact(out[key]);
      }
    }
    return out;
  };
  if (event.request) event.request = redact(event.request);
  if (event.user) event.user = { ...event.user, ip_address: undefined, email: undefined };
  if (event.extra) event.extra = redact(event.extra);
  if (event.contexts) event.contexts = redact(event.contexts);
  return event;
}
```

```ts
// app/api/sentry-smoke/route.ts
// D-19: Sentry smoke route. Throws ONLY when ?token=<SENTRY_SMOKE_TOKEN>.
// Returns 404 for unauthorised hits so crawlers / scanners cannot abuse it.
//
// Usage: after a deploy, hit
//   https://<preview-url>/api/sentry-smoke?token=<SENTRY_SMOKE_TOKEN>
// and confirm an issue titled "sentry-smoke — deliberate error" appears in Sentry within 60s.
import { NextResponse, type NextRequest } from 'next/server';

export const runtime = 'nodejs';
export const dynamic = 'force-dynamic'; // never cached — we always want the real branch

export async function GET(req: NextRequest) {
  const expected = process.env.SENTRY_SMOKE_TOKEN;
  const provided = req.nextUrl.searchParams.get('token');

  // No token configured on this env → pretend the route doesn't exist.
  if (!expected) {
    return new NextResponse('Not Found', { status: 404 });
  }

  // Timing-safe comparison so we don't leak the expected length via response time.
  if (!provided || !timingSafeEqual(provided, expected)) {
    return new NextResponse('Not Found', { status: 404 });
  }

  // Deliberate throw — server Sentry SDK captures it and reports to the issue feed.
  throw new Error('sentry-smoke — deliberate error');
}

// Small helper. Not imported from 'crypto' because Next's edge-safe nodejs runtime
// supports the Web Crypto subtle API for this check.
function timingSafeEqual(a: string, b: string): boolean {
  if (a.length !== b.length) return false;
  let mismatch = 0;
  for (let i = 0; i < a.length; i++) {
    mismatch |= a.charCodeAt(i) ^ b.charCodeAt(i);
  }
  return mismatch === 0;
}
```

Why `export const runtime = 'nodejs'`: default Next.js route handlers are Node — explicit avoids future drift if we flip to edge anywhere. Why `force-dynamic`: never cache a 404 vs 500 branch difference.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser client → Sentry SaaS | Client events leave the browser with the app's DSN. DSN is client-safe by design, but PII in events is not — scrubbed via sendDefaultPii + beforeSend |
| server / edge route → Sentry SaaS | Server events may carry request headers, env, user context — scrubbed via identical server-side beforeSend |
| public internet → `/api/sentry-smoke` | Any HTTP client can hit this route; must be gated by secret token (D-19) so it cannot be weaponised |
| Vercel build runner → Sentry auth token | SENTRY_AUTH_TOKEN is a privileged secret (can publish releases, upload source maps). Lives only in Vercel env. Never committed; Plan 02 gitleaks would catch a leak. |
