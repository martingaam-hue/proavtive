If `app/hk/holiday-camps/[slug]/page.tsx` does not exist yet: create a minimal dynamic camp page that fetches by slug and generates Event JSON-LD:

```typescript
// app/hk/holiday-camps/[slug]/page.tsx
import { notFound } from 'next/navigation'
import { sanityFetch } from '@/lib/sanity.live'
import { campBySlugQuery } from '@/lib/queries'
import { urlFor } from '@/components/sanity-image'

export default async function HKCampPage({ params }: { params: { slug: string } }) {
  const { data: camp } = await sanityFetch({
    query: campBySlugQuery,
    params: { slug: params.slug },
    tags: ['camp'],
  })

  if (!camp) notFound()

  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'Event',
    name: camp.title,
    description: camp.description ?? undefined,
    startDate: camp.startDate,
    endDate: camp.endDate,
    image: camp.image?.asset ? urlFor(camp.image.asset).width(1200).height(630).url() : undefined,
    location: camp.venue ? {
      '@type': 'Place',
      name: camp.venue.name,
      address: {
        '@type': 'PostalAddress',
        streetAddress: camp.venue.address,
        addressLocality: camp.venue.city,
        addressCountry: camp.venue.countryCode,
      },
    } : undefined,
    offers: camp.price ? {
      '@type': 'Offer',
      price: camp.price,
      priceCurrency: camp.priceCurrency ?? 'HKD',
      url: camp.offerUrl ?? undefined,
      availability: 'https://schema.org/InStock',
    } : undefined,
  }

  return (
    <>
      <script
        type="application/ld+json"
        dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
      />
      {/* Minimal camp detail: title, dates, description, booking CTA */}
      <main className="py-16">
        <div className="container mx-auto px-4">
          <h1 className="text-h1 font-display text-foreground mb-4">{camp.title}</h1>
          {camp.description && <p className="text-body text-muted-foreground mb-6">{camp.description}</p>}
          {camp.ageRange && <p className="text-label font-medium">Ages: {camp.ageRange}</p>}
          {camp.offerUrl && (
            <a href={camp.offerUrl} className="inline-flex items-center bg-brand-red text-white px-6 py-3 rounded-lg font-semibold mt-6">
              Book This Camp
            </a>
          )}
        </div>
      </main>
    </>
  )
}

export async function generateStaticParams() {
  const { data: slugs } = await sanityFetch({ query: import('@/lib/queries').then(m => m.hkBlogSlugsQuery) })
  // Use campBySlugQuery-equivalent slug list — fetch all camp slugs for hk
  return []  // Fallback: dynamic rendering; ISR handles revalidation
}
```

Note: For the camp `[slug]` page, if `generateStaticParams` is complex to set up initially, an empty return is acceptable — Next.js will render dynamically on first request and ISR will cache.
</task>
